    <script>
        const names = ['Palmetto Pines','Lakeview Estates','Greenwood Retreat','Blue Ridge Park','Smoky Valley RV','Sunset Acres','Cedar Creek','Magnolia Pines','River Bend RV','Lakeside Long Stay','Southern Oaks RV','Pineview Estates','Mountain Valley RV','Riverside Retreat','Creekside Park','Harvest Moon RV','Cardinal Ridge','Dogwood Acres','Cypress Bend RV','Heritage Pines','Trailhead RV','Riverbend Estates','Clover Hill RV','Autumn Ridge Park','Peach State RV'];
        const states = ['SC','GA','NC','TN'];
        const deals = names.map((name, i) => {
            const pads = 42 + Math.floor(i * 2.8);
            const price = 2500000 + i * 290000;
            const occ = 65 + Math.floor(i * 1.1);
            const curPad = 520 + (i % 5) * 20;
            const mktPad = curPad + 60 + (i % 4) * 15;
            const under = Math.round((1 - curPad / mktPad) * 100);
            const t12Inc = Math.round(pads * curPad * 12 * (occ / 100));
            const t12Exp = Math.round(t12Inc * (0.40 + (i % 5) * 0.02));
            const t12Noi = t12Inc - t12Exp;
            const curCap = +((t12Noi / price) * 100).toFixed(1);
            const stabInc = Math.round(pads * mktPad * 12 * 0.92);
            const stabNoi = Math.round(stabInc * 0.57);
            const proCap = +((stabNoi / price) * 100).toFixed(1);
            const spread = Math.round((proCap - curCap) * 100);
            return {
                id: 'DEAL-' + String(i+1).padStart(4,'0'), name,
                state: states[i % 4], pads,
                price: '$' + (price/1000000).toFixed(2) + 'M',
                cap: curCap.toFixed(1) + '%',
                occ: occ + '%',
                score: Math.min(95, 52 + Math.floor(i * 1.7)),
                status: i < 20 ? 'New' : i < 23 ? 'Shortlist' : 'Deep dive',
                under: under + '%',
                curCap: curCap.toFixed(1) + '%',
                proCap: proCap.toFixed(1) + '%',
                spread: '+' + spread + ' bps',
                t12Inc: '$' + t12Inc.toLocaleString(),
                t12Noi: '$' + t12Noi.toLocaleString(),
                totalPads: pads,
                expansion: (1.2 + i * 0.15).toFixed(1),
                breakEven: (52 + i % 15) + '%'
            };
        });
        let current = 'all';
        function render() {
            const tbody = document.getElementById('tbody');
            tbody.innerHTML = '';
            deals.filter(d => current === 'all' || d.status === current).forEach(d => {
                const badge = d.status === 'New' ? 'badge-new' : d.status === 'Shortlist' ? 'badge-shortlist' : 'badge-deepdive';
                const tr = document.createElement('tr');
                tr.innerHTML = `<td><b>${d.id}</b></td><td>${d.name}</td><td><b style="color:#059669">${d.state}</b></td><td><b>${d.pads}</b></td><td>${d.price}</td><td>${d.cap}</td><td>${d.occ}</td><td class="score">${d.score}</td><td><span class="badge ${badge}">${d.status}</span></td><td><button class="dive-btn" onclick="dive('${d.id}')">Deep Dive</button></td>`;
                tbody.appendChild(tr);
            });
        }
        function filter(btn, val) {
            current = val;
            document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            render();
        }
        function dive(id) {
            const d = deals.find(x => x.id === id);
            document.getElementById('panelTitle').textContent = d.name + ' | ' + d.state + ' | ' + d.pads + ' pads | ' + d.price;
            document.getElementById('topMetrics').innerHTML = `<div class="metric"><div class="metric-val green">${d.under}</div><div class="metric-lbl">Pad Rate Under Market</div></div><div class="metric"><div class="metric-val">${d.curCap}</div><div class="metric-lbl">Current Cap Rate</div></div><div class="metric"><div class="metric-val">${d.proCap}</div><div class="metric-lbl">Pro-forma Cap Rate</div></div><div class="metric"><div class="metric-val teal">${d.spread}</div><div class="metric-lbl">Cap Rate Spread</div></div>`;
            document.getElementById('bottomMetrics').innerHTML = `<div class="metric"><div class="metric-val">${d.t12Inc}</div><div class="metric-lbl">T-12 Income</div></div><div class="metric"><div class="metric-val">${d.t12Noi}</div><div class="metric-lbl">T-12 NOI</div></div><div class="metric"><div class="metric-val">${d.totalPads}</div><div class="metric-lbl">Total Pads</div></div><div class="metric"><div class="metric-val">${d.expansion}</div><div class="metric-lbl">Expansion Acres</div></div><div class="metric"><div class="metric-val">${d.breakEven}</div><div class="metric-lbl">Breakeven Occ</div></div>`;
            const panel = document.getElementById('panel');
            panel.style.display = 'block';
            panel.scrollIntoView({ behavior: 'smooth' });
        }
        function closePanel() { document.getElementById('panel').style.display = 'none'; }
        render();
    </script>
